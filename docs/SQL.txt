CREATE TABLE IF NOT EXISTS users (
    id INT AUTO_INCREMENT PRIMARY KEY,
    fingerprint_hash VARCHAR(64) UNIQUE NOT NULL,
    first_visit TIMESTAMP NOT NULL,
    last_activity TIMESTAMP NOT NULL,
    total_sessions INT NOT NULL DEFAULT 0,
    total_videos_processed INT NOT NULL DEFAULT 0,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

CREATE TABLE IF NOT EXISTS user_sessions (
    id INT AUTO_INCREMENT PRIMARY KEY,
    user_id INT NOT NULL,
    jwt_token_hash VARCHAR(64) UNIQUE NOT NULL,
    expires_at TIMESTAMP NOT NULL,
    user_agent TEXT NULL,
    ip_address VARCHAR(45) NULL,
    is_active BOOLEAN NOT NULL DEFAULT TRUE,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (user_id) REFERENCES users(id) ON DELETE CASCADE
);

CREATE TABLE IF NOT EXISTS video_sessions (
    id INT AUTO_INCREMENT PRIMARY KEY,
    user_id INT NOT NULL,
    video_url TEXT NOT NULL,
    processing_status ENUM('processing', 'completed', 'failed') NOT NULL DEFAULT 'processing',
    started_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    completed_at TIMESTAMP NULL,
    FOREIGN KEY (user_id) REFERENCES users(id) ON DELETE CASCADE
);

CREATE TABLE IF NOT EXISTS frame_annotations (
    id INT AUTO_INCREMENT PRIMARY KEY,
    video_session_id INT NOT NULL,
    frame_timestamp DECIMAL(10,3) NOT NULL,
    annotation_data JSON NOT NULL,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (video_session_id) REFERENCES video_sessions(id) ON DELETE CASCADE
);

CREATE INDEX idx_users_fingerprint ON users(fingerprint_hash);
CREATE INDEX idx_user_sessions_active ON user_sessions(user_id, is_active, expires_at);
CREATE INDEX idx_video_sessions_status ON video_sessions(user_id, processing_status);
CREATE INDEX idx_frame_annotations_video ON frame_annotations(video_session_id, frame_timestamp);
